expression = - exponential
    | ~ exponential
    | ! exponential
    | exponential

exponential = multiplicative ** exponential
    | multiplicative

multiplicative = multiplicative * arithmetic
    | multiplicative / arithmetic
    | multiplicative % arithmetic
    | arithmetic

arithmetic = arithmetic + bitwise
    | arithmetic - bitwise
    | bitwise

bitwise = bitwise & relational
    | bitwise | relational
    | bitwise ^ relational
    | bitwise << relational
    | bitwise >> relational
    | relational

relational = relational == logic
    | relational != logic
    | relational < logic
    | relational <= logic
    | relational >= logic
    | relational >= logic
    | logic

logic = logic && atom
    | logic || atom
    | atom

atom = INT
    | FLOAT
    | BOOLEAN
    | STRING
    | 
    | LEFT_PARENTHESIS expression RIGHT_PARENTHESIS
