    | arithmetic MULTIPLY bitwise
    | arithmetic DIVIDE bitwise
    | arithmetic MODULO bitwise

    | arithmetic SUBTRACT bitwise
    | NEGATE arithmetic

bitwise = bitwise BIT_AND relational
    | bitwise BIT_OR relational
    | bitwise BIT_XOR relational
    | bitwise BIT_NOT relational
    | bitwise BIT_LEFT_SHIFT relational
    | bitwise BIT_RIGHT_SHIFT relational
    | relational

relational = relational EQUALS logic
    | relational NOT_EQUALS logic
    | relational LESS_THAN logic
    | relational LESS_THAN_EQUAL logic
    | relational GREATER_THAN logic
    | relational GREATER_THAN_EQUAL logic
    | logic

logic = logic AND atom
    | logic OR atom
    | NOT atom
    | atom

atom = INT
    | FLOAT
    | BOOLEAN
    | STRING
    | CELL_ADDRESS
